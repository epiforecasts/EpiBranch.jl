Line list and contacts generation

Reproducing the simulist R package functionality

Overview

This vignette demonstrates generating simulated line lists and contact tracing data, reproducing the core functionality of the simulist R package.

using EpiBranch
using Distributions
using DataFrames
using Dates
using Random: Xoshiro

Basic line list

Simulate an outbreak and generate a line list with demographics, delays, and outcomes:

model = BranchingProcess(Poisson(1.5), LogNormal(1.6, 0.5))

rng = Xoshiro(42)
state = simulate_conditioned(model, 50:200;
    sim_opts = SimOpts(
        max_cases = 200,
        incubation_period = LogNormal(1.5, 0.5),
    ),
    rng = rng
)

rng2 = Xoshiro(99)
ll = linelist(state;
    reference_date = Date(2024, 1, 1),
    delay_opts = DelayOpts(
        onset_to_reporting = Exponential(3.0),
        onset_to_admission = Exponential(5.0),
        onset_to_outcome = Exponential(14.0),
    ),
    outcome_opts = OutcomeOpts(
        prob_hospitalisation = 0.2,
        prob_death = 0.05,
    ),
    rng = rng2,
)

println("Line list: $(nrow(ll)) cases, $(ncol(ll)) columns")
first(ll, 5)

Demographics

Control the age distribution and sex ratio:

rng3 = Xoshiro(99)
ll_demo = linelist(state;
    reference_date = Date(2024, 1, 1),
    demographic_opts = DemographicOpts(
        age_distribution = Normal(40, 15),
        age_range = (0, 90),
        prob_female = 0.55,
    ),
    rng = rng3,
)

println("Age range: $(minimum(ll_demo.age)) - $(maximum(ll_demo.age))")
println("Female proportion: $(round(count(==("female"), ll_demo.sex) / nrow(ll_demo), digits=2))")

Age-stratified case fatality risk

age_cfr = Dict(
    (0, 14) => 0.001,
    (15, 64) => 0.01,
    (65, 90) => 0.15,
)

rng4 = Xoshiro(99)
ll_age_cfr = linelist(state;
    reference_date = Date(2024, 1, 1),
    outcome_opts = OutcomeOpts(age_specific_cfr = age_cfr),
    demographic_opts = DemographicOpts(age_range = (0, 90)),
    rng = rng4,
)

# Deaths by age group
for (lo, hi) in [(0, 14), (15, 64), (65, 90)]
    group = filter(r -> lo <= r.age <= hi, ll_age_cfr)
    n_died = count(==("died"), group.outcome)
    println("Age $lo-$hi: $(nrow(group)) cases, $n_died deaths ($(round(n_died/max(nrow(group),1)*100, digits=1))%)")
end

Contacts table

The contacts table includes both infected and non-infected contacts, with a was_case flag:

ct = contacts(state; reference_date = Date(2024, 1, 1))
println("Total contacts: ", nrow(ct))
println("Infected contacts (was_case=true): ", count(ct.was_case))
println("Non-infected contacts: ", count(.!ct.was_case))

Conditioning on outbreak size

Use rejection sampling to generate outbreaks of a specific size:

rng = Xoshiro(42)
state = simulate_conditioned(model, 30:100;
    sim_opts = SimOpts(
        max_cases = 200,
        incubation_period = LogNormal(1.5, 0.5),
    ),
    rng = rng,
)

println("Outbreak size: ", state.cumulative_cases, " (target: 30-100)")
Outbreak size: 33 (target: 30-100)

Combined outbreak simulation

Generate both line list and contacts from the same simulation:

rng = Xoshiro(42)
model = BranchingProcess(NegBin(1.5, 0.5), LogNormal(1.6, 0.5))
state = simulate_conditioned(model, 50:200;
    sim_opts = SimOpts(
        max_cases = 200,
        n_initial = 3,
        incubation_period = LogNormal(1.5, 0.5),
    ),
    rng = rng,
)

ref_date = Date(2024, 1, 1)
ll = linelist(state; reference_date = ref_date, rng = Xoshiro(99))
ct = contacts(state; reference_date = ref_date)
cs = chain_statistics(state)

println("Line list: $(nrow(ll)) cases")
println("Contacts: $(nrow(ct)) contact events ($(count(ct.was_case)) infected)")
println("Chains: $(nrow(cs)) chains")
println("Chain sizes: ", cs.size)

Chains: 3 chains
